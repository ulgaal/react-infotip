Location

import React from 'react' import PropTypes from 'prop-types' import { LocationType } from './prop-types'

Location is an internal component used to position a tooltip with respect to its target. It uses absolute positioning and a CSS translate() transform.

const Location = props => { const { onMouseLeave, location: { left, top }, children } = props return ( <span onMouseLeave={onMouseLeave} className='rit-location' style={{ position: 'absolute', display: 'inline-block', zIndex: 100, transform: `translate(${left}px, ${top}px)`, left: 0, top: 0 }} > {children} </span> ) } Location.propTypes = { location: LocationType, onMouseLeave: PropTypes.func } export default Location
